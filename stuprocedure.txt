Step 2: Write the Smart Contract Code

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.8;

contract StudentData {
    
    struct Student {
        uint256 rollNo;
        string name;
        string department;
        uint256 marks;
    }

    Student public student;

    // Function to set student details
    function setStudentData(uint256 _rollNo, string memory _name, string memory _department, uint256 _marks) public {
        student = Student(_rollNo, _name, _department, _marks);
    }

    // Function to get student details
    function getStudentData() public view returns (uint256, string memory, string memory, uint256) {
        return (student.rollNo, student.name, student.department, student.marks);
    }
}

Optional Enhancement (for multiple students)

If you want to store many students instead of just one, use an array or mapping:

pragma solidity ^0.8.8;

contract StudentRecords {
    struct Student {
        uint256 rollNo;
        string name;
        string department;
        uint256 marks;
    }

    Student[] public students;

    function addStudent(uint256 _rollNo, string memory _name, string memory _department, uint256 _marks) public {
        students.push(Student(_rollNo, _name, _department, _marks));
    }

    function getStudent(uint256 index) public view returns (uint256, string memory, string memory, uint256) {
        Student memory s = students[index];
        return (s.rollNo, s.name, s.department, s.marks);
    }
}
